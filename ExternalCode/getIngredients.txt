	}
	
	//select db to use
	$db_select = mysql_select_db("mybar",$connection);
	if(!$db_select)
	{
		die("database selection failed" . mysql_error());
	}
	
	$result = mysql_query("SELECT * FROM ingredient", $connection);

	$convert = array("ingredients" => array());

	while($row = mysql_fetch_array($result)){
		$jsonRow = array(
			"_id" => $row[IID],
			"name" => $row[NAME],
			"url" => $row[PICTUREURL],
			"alcohollevel" => $row[ALCOHOLLEVEL],
			"description" => $row[DESCRIPTION],
			"ingredienttype" => $row[INGREDIENTTYPE]
			);
		array_push($convert["ingredients"],$jsonRow);
	}
	
	echo json_encode($convert);
